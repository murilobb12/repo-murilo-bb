 package org.example.service;
 
 import org.example.model.Desempenho;
 import org.example.model.Funcionario;
 
 import java.math.BigDecimal;
 import java.math.RoundingMode;
 
 public class FuncionarioService {
 
     public BigDecimal calcularBonus(Funcionario funcionario) {
 
         BigDecimal valor = funcionario.getSalario().multiply(new BigDecimal("0.1"));
         if (valor.compareTo(new BigDecimal("1000")) > 0) {
             valor = BigDecimal.ZERO;
         }
         return valor.setScale(2, RoundingMode.HALF_UP);
 
     }
 
     public BigDecimal reajusteSalarial(Funcionario funcionario) {
 
 //        if (funcionario.getDesempenho()==null){
 //            throw new InvalidParameterException("Deparamento do funcionario nao informado.");
 //        }
 
 
 
             switch (funcionario.getDesempenho()) {
                 case BOM:
                     return funcionario.getSalario().multiply(Desempenho.BOM.percentualReajuste()).setScale(2, RoundingMode.HALF_UP);
                 case OTIMO:
                     return funcionario.getSalario().multiply(Desempenho.OTIMO.percentualReajuste()).setScale(2, RoundingMode.HALF_UP);
                 case A_DESEJAR:
                     return funcionario.getSalario().multiply(Desempenho.A_DESEJAR.percentualReajuste()).setScale(2, RoundingMode.HALF_UP);
                 default:
                     return null;
             }
 
 
     }
 
 }
